      document.querySelectorAll('.room-card button').forEach(button => {
          button.addEventListener('click', () => {
              alert('Room booked! Please check your account for details.');
          });
      });

      window.addEventListener('scroll', () => {
          const myBookings = document.querySelector('.my-bookings');
          const scrollPosition = window.scrollY;
          const triggerPosition = document.querySelector('.rooms').offsetTop + document.querySelector('.rooms').offsetHeight;

          if (scrollPosition >= triggerPosition) {
              myBookings.classList.add('active');
          } else {
              myBookings.classList.remove('active');
          }
      });

      document.querySelector('.cancel').addEventListener('click', () => {

      });
